    <script>
        (function() {
                // ---------- MOCK STORAGE (simulate DB) ----------
                const MockDB = {
                    users: [{
                        id: '1',
                        name: 'Admin One',
                        email: 'admin@demo.com',
                        password: 'admin123',
                        role: 'admin'
                    }, {
                        id: '2',
                        name: 'John Trader',
                        email: 'john@demo.com',
                        password: 'user123',
                        role: 'user'
                    }],
                    transactions: [{
                        id: 't1',
                        userId: '2',
                        userEmail: 'john@demo.com',
                        symbol: 'AAPL',
                        type: 'BUY',
                        shares: 10,
                        price: 175.23,
                        status: 'pending'
                    }, {
                        id: 't2',
                        userId: '2',
                        userEmail: 'john@demo.com',
                        symbol: 'TSLA',
                        type: 'SELL',
                        shares: 2,
                        price: 245.67,
                        status: 'approved'
                    }, ],
                    watchlists: {
                        '2': ['AAPL', 'MSFT']
                    }, // userId -> symbols
                    getNextUserId() {
                        return String(this.users.length + 1);
                    },
                    getNextTxId() {
                        return String(this.transactions.length + 1);
                    }
                };

                // ---------- FAKE STOCK API (real-time simulation) ----------
                const stockCache = new Map();

                function fetchStockQuote(symbol) {
                    return new Promise((resolve) => {
                        setTimeout(() => {
                            const base = stockCache.get(symbol) || {
                                price: 100 + Math.random() * 200,
                                high: 0,
                                low: 0,
                                prev: 0
                            };
                            // simulate variation
                            const changePercent = (Math.random() * 6 - 3) / 100; // -3% .. +3%
                            const newPrice = base.price * (1 + changePercent);
                            const quote = {
                                symbol: symbol.toUpperCase(),
                                price: Math.round(newPrice * 100) / 100,
                                high: Math.round(newPrice * 1.02 * 100) / 100,
                                low: Math.round(newPrice * 0.98 * 100) / 100,
                                prev: Math.round(base.price * 100) / 100,
                                change: (newPrice - base.price).toFixed(2),
                                changePercent: (changePercent * 100).toFixed(2)
                            };
                            stockCache.set(symbol, {
                                price: newPrice,
                                high: quote.high,
                                low: quote.low,
                                prev: base.price
                            });
                            resolve(quote);
                        }, 300);
                    });
                }

                // chart history mock (7 days)
                function getHistory() {
                    return [22, 25, 23, 28, 30, 27, 29].map(v => v + (Math.random() * 4 - 2));
                }

                // ---------- CONTEXT SIMULATION (global state) ----------
                let authState = {
                    user: null, // { id, name, email, role }
                    token: null,
                    watchlistCount: 0
                };
                let theme = 'light'; // 'light' or 'dark'

                const listeners = new Set();

                function notify() {
                    listeners.forEach(fn => fn());
                }

                // AuthContext
                function setAuth(user, token) {
                    authState.user = user;
                    authState.token = token;
                    if (user) {
                        const wl = MockDB.watchlists[user.id] || [];
                        authState.watchlistCount = wl.length;
                    } else {
                        authState.watchlistCount = 0;
                    }
                    notify();
                }

                function login(email, password) {
                    return new Promise((resolve, reject) => {
                        setTimeout(() => {
                            const user = MockDB.users.find(u => u.email === email && u.password === password);
                            if (user) {
                                const token = `mock-jwt-${user.id}-${Date.now()}`;
                                setAuth({
                                    id: user.id,
                                    name: user.name,
                                    email: user.email,
                                    role: user.role
                                }, token);
                                resolve({
                                    success: true,
                                    user
                                });
                            } else reject('Invalid credentials');
                        }, 400);
                    });
                }

                function signup(name, email, password) {
                    return new Promise((resolve, reject) => {
                        setTimeout(() => {
                            if (MockDB.users.find(u => u.email === email)) return reject('Email exists');
                            const newUser = {
                                id: MockDB.getNextUserId(),
                                name,
                                email,
                                password,
                                role: 'user'
                            };
                            MockDB.users.push(newUser);
                            MockDB.watchlists[newUser.id] = [];
                            resolve({
                                success: true
                            });
                        }, 400);
                    });
                }

                function logout() {
                    setAuth(null, null);
                }

                // watchlist helpers
                function addToWatchlist(symbol) {
                    if (!authState.user) return;
                    const uid = authState.user.id;
                    let list = MockDB.watchlists[uid] || [];
                    if (!list.includes(symbol)) {
                        list.push(symbol);
                        MockDB.watchlists[uid] = list;
                        authState.watchlistCount = list.length;
                        notify();
                    }
                }

                function removeFromWatchlist(symbol) {
                    if (!authState.user) return;
                    const uid = authState.user.id;
                    let list = MockDB.watchlists[uid] || [];
                    list = list.filter(s => s !== symbol);
                    MockDB.watchlists[uid] = list;
                    authState.watchlistCount = list.length;
                    notify();
                }

                // transactions
                function createTransaction(type, symbol, shares, price) {
                    if (!authState.user) return;
                    const newTx = {
                        id: MockDB.getNextTxId(),
                        userId: authState.user.id,
                        userEmail: authState.user.email,
                        symbol: symbol.toUpperCase(),
                        type,
                        shares,
                        price,
                        status: 'pending'
                    };
                    MockDB.transactions.push(newTx);
                    notify();
                }

                // admin approve/reject
                function updateTxStatus(txId, status) {
                    const tx = MockDB.transactions.find(t => t.id === txId);
                    if (tx) {
                        tx.status = status;
                        notify();
                    }
                }

                // ---------- ThemeContext ----------
                function toggleTheme() {
                    theme = theme === 'light' ? 'dark' : 'light';
                    document.body.className = theme;
                    notify();
                }

                // ---------- React-like render ----------
                function renderApp() {
                    const root = document.getElementById('app');
                    if (!authState.user) {
                        root.innerHTML = renderAuthPages();
                    } else {
                        root.innerHTML = renderDashboard();
                    }
                    attachEvents();
                }

                listeners.add(renderApp);

                // ---------- AUTH PAGES (login/signup) ----------
                let authPage = 'login'; // 'login' or 'signup'
                function renderAuthPages() {
                    return `
          <div class="auth-form">
            <h2 style="margin-bottom:1.5rem;">📊 StockDash · ${authPage === 'login' ? 'Login' : 'Sign up'}</h2>
            ${authPage === 'login' ? `
              <input type="email" id="loginEmail" placeholder="Email" value="john@demo.com" />
              <input type="password" id="loginPassword" placeholder="Password" value="user123" />
              <button class="btn btn-primary" style="width:100%;" id="loginBtn">Log in</button>
              <p class="toggle-link" id="switchToSignup">Create account →</p>
            ` : `
              <input type="text" id="signupName" placeholder="Full name" />
              <input type="email" id="signupEmail" placeholder="Email" />
              <input type="password" id="signupPassword" placeholder="Password" />
              <button class="btn btn-primary" style="width:100%;" id="signupBtn">Sign up</button>
              <p class="toggle-link" id="switchToLogin">← Back to login</p>
            `}
            <div id="authMessage" class="message hidden"></div>
          </div>
        `;
      }

      // ---------- DASHBOARD (main app) ----------
      function renderDashboard() {
        const user = authState.user;
        const isAdmin = user?.role === 'admin';
        const wl = MockDB.watchlists[user.id] || [];
        // fetch current stock? We'll render static then dynamic update
        return `
          <div class="navbar">
            <div class="logo"><h2>📈 <span>Stock</span>Dash</h2></div>
            <div class="nav-items">
              <span><i class="fas fa-chart-line"></i> Markets</span>
              <span>Watchlist <span class="watchlist-badge" id="watchlistCount">${authState.watchlistCount}</span></span>
              <div class="user-info">
                <span>${user.name}</span> <div class="avatar">${user.name[0]}</div>
                <button class="theme-toggle" id="themeToggleBtn"><i class="fas ${theme==='dark' ? 'fa-sun' : 'fa-moon'}"></i></button>
                <i class="fas fa-sign-out-alt" id="logoutBtn" style="cursor:pointer;"></i>
              </div>
            </div>
          </div>

          <div class="panel-tabs">
            <button class="tab-btn active" id="tabDashboard">Dashboard</button>
            ${isAdmin ? '<button class="tab-btn" id="tabAdmin">Admin Panel</button>' : '<button class="tab-btn" id="tabUser">My Trades</button>'}
          </div>

          <div id="dynamicPanel">
            ${renderMainPanel()}
          </div>
        `;
      }

      function renderMainPanel() {
        const user = authState.user;
        const isAdmin = user?.role === 'admin';
        if (isAdmin) return renderAdminPanel();
        else return renderUserPanel();
      }

      // ----- EVENT HANDLING + DYNAMIC UPDATES -----
      function attachEvents() {
        // theme toggle
        document.getElementById('themeToggleBtn')?.addEventListener('click', toggleTheme);
        document.getElementById('logoutBtn')?.addEventListener('click', () => logout());

        // auth switching
        document.getElementById('switchToSignup')?.addEventListener('click', (e) => {
          e.preventDefault(); authPage = 'signup'; renderApp();
        });
        document.getElementById('switchToLogin')?.addEventListener('click', (e) => {
          e.preventDefault(); authPage = 'login'; renderApp();
        });

        // login / signup actions
        document.getElementById('loginBtn')?.addEventListener('click', async () => {
          const email = document.getElementById('loginEmail').value;
          const pass = document.getElementById('loginPassword').value;
          try { await login(email, pass); } catch (err) { showMessage('authMessage', err); }
        });
        document.getElementById('signupBtn')?.addEventListener('click', async () => {
          const name = document.getElementById('signupName').value;
          const email = document.getElementById('signupEmail').value;
          const pass = document.getElementById('signupPassword').value;
          try { await signup(name, email, pass); authPage='login'; renderApp(); } catch (err) { showMessage('authMessage', err); }
        });

        // dashboard tabs
        document.getElementById('tabDashboard')?.addEventListener('click', () => {
          document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
          document.getElementById('tabDashboard')?.classList.add('active');
          document.getElementById('dynamicPanel').innerHTML = renderUserPanel();  // default
          attachPanelEvents();
        });
        document.getElementById('tabAdmin')?.addEventListener('click', () => {
          document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
          document.getElementById('tabAdmin')?.classList.add('active');
          document.getElementById('dynamicPanel').innerHTML = renderAdminPanel();
          attachAdminEvents();
        });
        document.getElementById('tabUser')?.addEventListener('click', () => {
          document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
          document.getElementById('tabUser')?.classList.add('active');
          document.getElementById('dynamicPanel').innerHTML = renderUserPanel();
          attachPanelEvents();
        });

        // stock search, buy, watchlist etc (user panel)
        attachPanelEvents();
        attachAdminEvents();
      }

      function attachPanelEvents() {
        const searchBtn = document.getElementById('searchStockBtn');
        const symbolInp = document.getElementById('stockSymbolInput');
        const quoteDiv = document.getElementById('stockQuoteArea');
        let currentQuote = null;

        if (searchBtn) {
          searchBtn.addEventListener('click', async () => {
            const sym = symbolInp.value.trim().toUpperCase();
            if (!sym) return;
            const quote = await fetchStockQuote(sym);
            currentQuote = quote;
            quoteDiv.innerHTML = `
              <div style="display:flex; align-items:baseline; gap:1rem;">
                <span class="price-big">$${quote.price}</span>
                <span class="${quote.changePercent >=0 ? 'positive' : 'negative'}">
                  ${quote.changePercent}% (${quote.change})
                </span>
              </div>
              <div class="stock-meta">
                <span>📊 H: $${quote.high}</span> <span>📉 L: $${quote.low}</span> <span>🔙 Prev: $${quote.prev}</span>
              </div>
            `;
          });
        }

        document.getElementById('buyStockBtn')?.addEventListener('click', () => {
          if (!currentQuote) return alert('Search stock first');
          const shares = prompt('Shares to buy?', '5');
          if (shares) {
            createTransaction('BUY', currentQuote.symbol, Number(shares), currentQuote.price);
            showMessage('tradeMessage', 'Buy request submitted (pending)');
          }
        });

        document.getElementById('sellStockBtn')?.addEventListener('click', () => {
          if (!currentQuote) return alert('Search stock first');
          const shares = prompt('Shares to sell?', '3');
          if (shares) {
            createTransaction('SELL', currentQuote.symbol, Number(shares), currentQuote.price);
            showMessage('tradeMessage', 'Sell request submitted');
          }
        });

        document.getElementById('addWatchlistBtn')?.addEventListener('click', () => {
          if (!currentQuote) return alert('Search stock first');
          addToWatchlist(currentQuote.symbol);
          renderApp(); // refresh
        });

        document.querySelectorAll('.remove-watch').forEach(el => {
          el.addEventListener('click', (e) => {
            const sym = e.target.dataset.sym;
            removeFromWatchlist(sym);
            renderApp();
          });
        });

        // draw dummy chart
        const ctx = document.getElementById('priceChart')?.getContext('2d');
        if (ctx) {
          new Chart(ctx, {
            type: 'line',
            data: { labels: ['D-6','D-5','D-4','D-3','D-2','D-1','Today'], datasets: [{ label: 'Price', data: getHistory(), borderColor: '#3b82f6' }] }
          });
        }
      }

      function attachAdminEvents() {
        document.querySelectorAll('.fa-check-circle').forEach(icon => {
          icon.addEventListener('click', (e) => {
            const txid = e.target.dataset.txid; updateTxStatus(txid, 'approved'); renderApp();
          });
        });
        document.querySelectorAll('.fa-times-circle').forEach(icon => {
          icon.addEventListener('click', (e) => {
            const txid = e.target.dataset.txid; updateTxStatus(txid, 'rejected'); renderApp();
          });
        });
      }

      function showMessage(elId, text) {
        const msg = document.getElementById(elId);
        if (msg) { msg.classList.remove('hidden'); msg.innerText = text; setTimeout(() => msg.classList.add('hidden'), 2500); }
      }

      // initial render
      renderApp();
    })();
